.MODEL SMALL
.STACK 1000H
.DATA
    number DB "00000$"
    w DW 10 DUP (0000H)    ; Line 1
.CODE

main PROC
    ;initialize data segment
    MOV AX, @DATA
    MOV DS, AX
	;preserve base pointer
    PUSH BP
    MOV BP, SP
    SUB SP, 2    ; Line 4
	L0:
    MOV BX, 0    ; Line 21
    MOV AX, [BP - 0]    ; Line 21
    CMP AX, BX    ; Line 21
	JG L1    ; Line 21
    JMP L2
	L4:
    MOV BX, 5    ; Line 21
    MOV AX, [BP - 0]    ; Line 21
    CMP AX, BX    ; Line 21
	JL L0    ; Line 21
    JMP L3
	;DEBUG: changing label: L5 ,to: L0

	;DEBUG: changing label: L6 ,to: L3

	L8:
    MOV BX, 10    ; Line 21
    MOV AX, [BP - 0]    ; Line 21
    CMP AX, BX    ; Line 21
	JL L1    ; Line 21
    JMP L4
    MOV AX, 15    ; Line 21
    NEG AX    ; Line 21
    PUSH AX    ; Line 21
	L12:
    POP BX    ; Line 21
    MOV AX, [BP - 0]    ; Line 21
    CMP AX, BX    ; Line 21
	JG L8    ; Line 21
    JMP L11
	;DEBUG: changing label: L13 ,to: L8

	;DEBUG: changing label: L14 ,to: L11

	;DEBUG: changing label: L9 ,to: L1

	;DEBUG: changing label: L10 ,to: L4

    L1:
    MOV AX, 1    ; Line 21
    MOV [BP - 0], AX    ; Line 21
    JMP L3
    L2:
    MOV AX, 0    ; Line 21
    MOV [BP - 0], AX    ; Line 21
    L3:
    MOV AX, [BP - 0]    ; Line 22
    CALL print_output    ; Line 22
    CALL new_line    ; Line 22
	;restore stack pointer
    ADD SP, 2
	;restore base pointer
    POP BP
    ;exit program
    MOV AX,4CH
    INT 21H
main ENDP


new_line proc
    push ax
    push dx
    mov ah,2
    mov dl,0Dh
    int 21h
    mov ah,2
    mov dl,0Ah
    int 21h
    pop dx
    pop ax
    ret
new_line endp
print_output proc  ;print what is in ax
    push ax
    push bx
    push cx
    push dx
    push si
    lea si,number
    mov bx,10
    add si,4
    cmp ax,0
    jnge negate
print:
    xor dx,dx
    div bx
    mov [si],dl
    add [si],'0'
    dec si
    cmp ax,0
    jne print
    inc si
    lea dx,si
    mov ah,9
    int 21h
    pop si
    pop dx
    pop cx
    pop bx
    pop ax
    ret
negate:
    push ax
    mov ah,2
    mov dl,'-'
    int 21h
    pop ax
    neg ax
    jmp print
print_output endp

END
